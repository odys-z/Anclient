project(album_tests LANGUAGES CXX)

FetchContent_Declare(
  googletest
  GIT_REPOSITORY https://github.com/google/googletest.git
  GIT_TAG        v1.15.2
)
FetchContent_MakeAvailable(googletest)

include(GoogleTest)

add_executable(album_tests
    ../app/src/wsclients.h
    ../app/src/wsclients.cpp
    ../app/src/doctier.cpp
    ../app/src/doclientier.cpp
    ../app/src/slingleton.cpp
    test_01_wsclient_ipcagent.cpp
    test_02_dequeue.cpp
    test_common.h
    test_settings.h
)

target_link_libraries(album_tests PRIVATE
    OpenSSL::Crypto
    Boost::url
    EnTT::EnTT
    nlohmann_json::nlohmann_json
    ixwebsocket::ixwebsocket
    anson
    anclient
    Slint::Slint
    album_slint_obj
    webview::core
    gtest_main
)

target_include_directories(album_tests PRIVATE
    ${CMAKE_SOURCE_DIR}/app/src
    ${CMAKE_BINARY_DIR}/app  # app-window.h is generated by Slint
)

# Post-build copy (resources)
add_custom_command(TARGET album_tests POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
        "${CMAKE_SOURCE_DIR}/ast"
        "$<TARGET_FILE_DIR:album_tests>/ast"

    COMMAND ${CMAKE_COMMAND} -E copy_directory
        "${CMAKE_SOURCE_DIR}/tests/settings"   # or move settings to root
        "$<TARGET_FILE_DIR:album_tests>/settings"

    COMMAND ${CMAKE_COMMAND} -E copy_directory
        "${CMAKE_SOURCE_DIR}/tests/res"
        "$<TARGET_FILE_DIR:album_tests>/res"

    COMMENT "Copying resources to build directory"
)

add_compile_definitions(ANSON_VERBOSE=4)

# Windows DLLs
if(WIN32)
    add_custom_command(TARGET album_tests POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy
            $<TARGET_RUNTIME_DLLS:album_tests>
            $<TARGET_FILE_DIR:album_tests> COMMAND_EXPAND_LISTS)
endif()

gtest_discover_tests(album_tests
    DISCOVERY_MODE PRE_TEST
)
